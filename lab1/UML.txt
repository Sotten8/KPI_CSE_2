enum Gender {
  MALE
  FEMALE
}

entity Owner {
  *owner_id : UUID
  --
  +name : string
  +surname : string
  +contact_data : text
}

entity Pet {
  *pet_id : UUID
  --
  +name : string
  +age : int
  +species : string
  +breed : string
  +gender : Gender
}

entity Vet {
  *vet_id : UUID
  --
  +name : string
  +surname : string
  +experience : string
  +contact_data : text
  +specialisation : string
}

entity Appointment {
  *timestamp : datetime
  *pet_id : UUID
  *vet_id : UUID
  --
  +reason : string
}

entity Diagnosis {
  *diagnosis_id : UUID
  --
  +is_healthy : boolean
  +description : text
}
